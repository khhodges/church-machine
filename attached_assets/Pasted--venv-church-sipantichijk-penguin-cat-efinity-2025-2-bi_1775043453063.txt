#!/bin/bash

# This script sets up the Efinity GUI environment variables
# It is expected to be located in the Efinix release bin directory
SCRIPT_PATH=${BASH_SOURCE[0]:-${(%):-%x}}
export EFINITY_HOME="$(cd "$(dirname "$SCRIPT_PATH")/.." && pwd)"
export EFXPT_HOME=$EFINITY_HOME/pt
export EFXPGM_HOME=$EFINITY_HOME/pgm
export EFXDBG_HOME=$EFINITY_HOME/debugger
export EFXIPM_HOME=$EFINITY_HOME/ipm
export EFXIPMGR_HOME=$EFINITY_HOME/ipm/bin/ip_manager
export EFXIPPKG_HOME=$EFINITY_HOME/ipm/bin/ip_packager
export EFXSVF_HOME=$EFINITY_HOME/debugger/svf_player
export EFXSERDESDBG_HOME=$EFINITY_HOME/debugger/serdes_debug_tool
export EFINITY_USER_DIR_INI=$HOME/.local/share/efinity/user_dir.ini

export QT_LOGGING_CONF=$EFINITY_HOME/bin/lc.ini
export QT_PLUGIN_PATH=$EFINITY_HOME/lib/plugins

# Disable Python user site support so we don't have package collisions
export PYTHONNOUSERSITE=1

# Issue a warning if the system GLIBC is below the minimum required version
if [[ "${OS:-}" != "Windows_NT" ]]; then
    EFX_REQUIRED_GLIBC_MAJOR=2
    EFX_REQUIRED_GLIBC_MINOR=28

    SYSTEM_GLIBC_VERSION_STRING=$(ldd --version | head -n 1)
    SYSTEM_GLIBC_VERSION_PATTERN=".*\(.*(GLIBC|GNU libc).*\) ([0-9]+)\.([0-9]+).*"

    if echo $SYSTEM_GLIBC_VERSION_STRING | grep -Eq "$SYSTEM_GLIBC_VERSION_PATTERN"; then
        SYSTEM_GLIBC_MAJOR=$(echo $SYSTEM_GLIBC_VERSION_STRING | sed -E "s/$SYSTEM_GLIBC_VERSION_PATTERN/\2/")
        SYSTEM_GLIBC_MINOR=$(echo $SYSTEM_GLIBC_VERSION_STRING | sed -E "s/$SYSTEM_GLIBC_VERSION_PATTERN/\3/")

        SYSTEM_GLIBC_VERSION_SUFFICIENT=1

        if (( SYSTEM_GLIBC_MAJOR < EFX_REQUIRED_GLIBC_MAJOR )); then
            SYSTEM_GLIBC_VERSION_SUFFICIENT=0
        elif (( SYSTEM_GLIBC_MAJOR == EFX_REQUIRED_GLIBC_MAJOR )) && (( SYSTEM_GLIBC_MINOR < EFX_REQUIRED_GLIBC_MINOR )); then
            SYSTEM_GLIBC_VERSION_SUFFICIENT=0
        fi

        if (( SYSTEM_GLIBC_VERSION_SUFFICIENT == 0 )); then 
            echo "WARNING: The system GLIBC version (${SYSTEM_GLIBC_MAJOR}.${SYSTEM_GLIBC_MINOR}) is below" \
                "the minimum required version (${EFX_REQUIRED_GLIBC_MAJOR}.${EFX_REQUIRED_GLIBC_MINOR})." \
                "The Efinity software may not function correctly on this system."
            echo ""
        fi
    else
        echo "WARNING: Failed to determine system GLIBC version. The Efinity software may not function correctly on this system."
        echo ""
    fi
fi

# Modify USER PATH to include EFINITY_HOME/bin
export PATH=$EFINITY_HOME/bin:$EFINITY_HOME/scripts:$EFXPGM_HOME/bin:$EFXDBG_HOME/bin:$EFXSVF_HOME/bin:$EFXIPM_HOME/bin:$EFXIPMGR_HOME:$EFXIPPKG_HOME:$EFXSERDESDBG_HOME/bin:$PATH
if [[ "${OS:-}" == "Windows_NT" ]]; then
    export PATH=$EFINITY_HOME/python311/bin:$PATH
    export QT_PLUGIN_PATH=$EFINITY_HOME/bin
        export PYTHONHOME=$EFINITY_HOME/python311
else
#    export LD_LIBRARY_PATH=$EFINITY_HOME/lib:$LD_LIBRARY_PATH
    export PYTHONPATH=$EFINITY_HOME/lib:$PYTHONPATH
        export PYTHONHOME=$EFINITY_HOME
    LINUX_DISTRO=""
    if [[ -f /etc/centos-release ]]; then
        LINUX_DISTRO="CentOS"
    elif [[ -f /etc/redhat-release ]]; then
        LINUX_DISTRO="RedHat"
    elif [[ -f /etc/os-release ]]; then
        LINUX_DISTRO=$(awk -F= '/^NAME/{print $2}' /etc/os-release)
    fi


    if [[ "${LINUX_DISTRO}" == '"Ubuntu"' ]] ; then
        # Disable QtWebengine sandbox support since it may requires root permission, and not supported in old kernel
        export QTWEBENGINE_DISABLE_SANDBOX=1
    elif [[ "${LINUX_DISTRO}" == "CentOS" ]] || [[ "${LINUX_DISTRO}" == "RedHat" ]]; then
        export QTWEBENGINE_DISABLE_SANDBOX=1
        export QTWEBENGINE_CHROMIUM_FLAGS="--disable-gpu"
    fi
fi
