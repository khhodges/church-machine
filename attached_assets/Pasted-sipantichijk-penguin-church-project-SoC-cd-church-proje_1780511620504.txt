sipantichijk@penguin:~/church_project/SoC$ cd ~/church_project/SoC
export EFINITY_HOME=~/efinity/2025.2 && source $EFINITY_HOME/bin/setup.sh
unset PYTHONHOME PYTHONPATH && export PATH=$EFINITY_HOME/bin:$PATH

# Copy new VDB for PnR:
cp top.vdb work_pnr/church_soc_cm.vdb

# Run PnR:
~/efinity/2025.2/bin/efx_pnr \
  --family Titanium --device Ti60F225 \
  --circuit church_soc_cm --prj church_soc_cm.xml \
  --pack --place --route \
  --max_threads 4 \
  --work_dir work_pnr \
  --output_dir outflow 2>&1 | tail -5

# Generate hex with BRAM init:
~/efinity/2025.2/bin/efx_pgm \
  --project church_soc_cm.xml \
  --source work_pnr/church_soc_cm.lbf \
  --dest outflow/church_soc_cm_church.hex \
  --enable_roms on 2>&1 | tail -5

# Check hex size — should be ~2.4MB (not ~2.16MB):
ls -la outflow/church_soc_cm_church.hex

# Flash and listen:
sudo /usr/bin/openFPGALoader -b titanium_ti60_f225_jtag -f outflow/church_soc_cm_church.hex && \
python3 -c "
import serial, time
s = serial.Serial('/dev/ttyUSB3', 115200, timeout=10, exclusive=False)
time.sleep(3); print(repr(s.read(500))); s.close()
"
